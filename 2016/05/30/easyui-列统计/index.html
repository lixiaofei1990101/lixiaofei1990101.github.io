easyui-列统计

2016-5-30 周一 16:03

easyui一些入门的东西，这里就不介绍了，百度一些，有很多，下载个API什么都能搞定。下面就easyui一个小功能的扩展谈一谈
有时候我们在做项目的过程中也许会针对那些数字的列进行统计，而且这样的情况出现还是比较的频繁的，当然啦，我们如果通过后台
的计算或者前台计算也可以实现这个功能，但是既然我们采用了easyui这个前端框架，那我们有什么好的办法没呢！废话不多说，直接贴
代码 

$.extend($.fn.datagrid.methods, {
    //仿照easyui的表格方法（例：appendRows的方法）
    statistics: function (jq) {
        //获取表格列
        var opt=$(jq).datagrid('options').columns;
        //取得表的行
        var rows = $(jq).datagrid("getRows");
        var footer = new Array();
        //定义统计的列有哪些
        footer['sum'] = "";
        //遍历表格的列，并判断是否存在统计的属性(sum)，并把我们需要统计的列提取出来
        for(var i=0; i<opt[0].length; i++){
            if(opt[0][i].sum){
                footer['sum'] = footer['sum'] + sum(opt[0][i].field)+ ',';
            }
        }
        //定义数组用于接收各列的统计值
        var footerObj = new Array();
        if(footer['sum'] != ""){
            var tmp = '{' + footer['sum'].substring(0,footer['sum'].length - 1) + "}";
            var obj = eval('(' + tmp + ')');
            if(obj[opt[0][1].field] == undefined){
                footer['sum'] += '"' + opt[0][1].field + '":""';
                obj = eval('({' + footer['sum'] + '})');
            }else{
                obj[opt[0][1].field] = "" + obj[opt[0][1].field];
            }
            footerObj.push(obj);
        }
        //将我们统计的值加载到页脚行（表格的最后一行）
        if(footerObj.length > 0){
            $(jq).datagrid('reloadFooter',footerObj); 
        }
        //将需要统计的列的值进行统计计算
        function sum(filed){
            var sumNum = 0;
            for(var i=0;i<rows.length;i++){
                sumNum += Number(rows[i][filed]);
            }
            //组装统计列的值，注：这里的sumNum可以根据你需要的数据，区别我们是否需要保留小数点sumNum.toFixed(2);
            return '"' + filed + '":"' + sumNum.toFixed(2) +'"';
        };
        //设置页脚的样式
        var panel = $(jq).datagrid('getPanel');
        var tr = panel.find('div.datagrid-footer tr'); 
        $(tr).children('td').css({"color":"red","font-size":"12px","font-weight":"bold"}) ;
    }
});

上面注释里面哟详细的实现步骤，在这里也就不多说了，这样以后，只需要按照下面的过程，表格加载完成后，就会自动帮我们进行列的统计了

1、给表格加上页脚的属性，用于存放我们统计的数据
    showFooter:true,
2、给字段加上sum的属性，作为统计的标识
    {field:'money',title:'交款金额',width:100,sum : true},
3、加载完成后进行统计
    $("#detailed_dg").datagrid({
      onLoadSuccess:function(){
         $("#detailed_dg").datagrid("statistics");
      },
   });

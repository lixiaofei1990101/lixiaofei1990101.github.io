JAVA中Excel的操作

2016-9-2 周五 11:15

在java这个行业里，主要有两个开源的工具用来操作Excel，一个是jxl，还有一个是poi。
jxl功能相对POI比较弱一点。但jExcelAPI对中文支持非常好，API是纯Java的
， 并不依赖Windows系统，即使运行在Linux下，它同样能够正确的处理Excel文件。 另外需要说明的是
，这套API对图形和图表的支持很有限，而且仅仅识别PNG格式。下面就是利用poi用于操作Excel编写的两个工具类。

导出Excel

在我们做java开发的过程中,导出Excel是经常会出现的，为了方便使用做成了一个通用的工具类，在这个工具中，我们只需要传入要导出数据的list集合对象和表头部分就OK了。
废话不多少，直接贴出代码。

package com.lixf.urp.unit;

import java.lang.reflect.Field;
import java.text.SimpleDateFormat;
import java.util.ArrayList;
import java.util.List;
import java.util.Map;
import java.util.Set;

import org.apache.poi.hssf.usermodel.HSSFWorkbook;
import org.apache.poi.ss.usermodel.Cell;
import org.apache.poi.ss.usermodel.Row;
import org.apache.poi.ss.usermodel.Sheet;
import org.apache.poi.ss.usermodel.Workbook;
import org.apache.poi.xssf.usermodel.XSSFWorkbook;

public class ExportExcel {

    public static final String VERSION_2003 = "2003";
    private static final int HEADER_ROW = 1;
    private static final int MAX_ROWS = 65535;
    private String data_format ="yyyy-MM-dd HH:mm:ss";

    private String version;
    private String[] sheetNames = new String[]{"sheet"};
    private int cellWidth = 8000;
    private int headerRow;
    private String[][] headers;
    private String[][] columns;

    private List<?>[] data;

    public ExportExcel dataFormat(String data_format) {
        this.data_format = data_format;
        return this;
    }

    public ExportExcel(List<?>... data) {
        this.data = data;
        this.column(new String[]{});
    }

    public static ExportExcel data(List<?>... data) {
        return new ExportExcel(data);
    }

    public static List<List<?>> dice(List<?> num, int chunkSize) {
        int size = num.size();
        int chunk_num = size / chunkSize + (size % chunkSize == 0 ? 0 : 1);
        List<List<?>> result = new ArrayList<List<?>>();
        for (int i = 0; i < chunk_num; i++) {
            result.add(num.subList(i * chunkSize, i == chunk_num - 1 ? size : (i + 1) * chunkSize));
        }
        return result;
    }

    @SuppressWarnings("unused")
    public Workbook export() throws Exception {
        Workbook wb;
        if (VERSION_2003.equals(version)) {
            wb = new HSSFWorkbook();
            if (data.length > 1) {
                for (int i = 0; i < data.length; i++) {
                    List<?> item = data[i];
                }
            } else if (data.length == 1 && data[0].size() > MAX_ROWS) {
                data = dice(data[0], MAX_ROWS).toArray(new List<?>[]{});
                String sheetName = sheetNames[0];
                sheetNames = new String[data.length];
                for (int i = 0; i < data.length; i++) {
                    sheetNames[i] = sheetName + (i == 0 ? "" : (i + 1));
                }
                String[] header = headers[0];
                headers = new String[data.length][];
                for (int i = 0; i < data.length; i++) {
                    headers[i] = header;
                }
                String[] column = columns[0];
                columns = new String[data.length][];
                for (int i = 0; i < data.length; i++) {
                    columns[i] = column;
                }
            }
        } else {
            wb = new XSSFWorkbook();
        }
        if (data.length == 0) {
            return wb;
        }
        for (int i = 0; i < data.length; i++) {
            Sheet sheet = wb.createSheet(sheetNames[i]);
            Row row;
            Cell cell;
            if (headers[i].length > 0) {
                row = sheet.createRow(0);
                if (headerRow <= 0) {
                    headerRow = HEADER_ROW;
                }
                headerRow = Math.min(headerRow, MAX_ROWS);
                for (int h = 0, lenH = headers[i].length; h < lenH; h++) {
                    if (cellWidth > 0) {
                        sheet.setColumnWidth(h, cellWidth);
                    }
                    cell = row.createCell(h);
                    cell.setCellValue(headers[i][h]);
                }
            }

            for (int j = 0, len = data[i].size(); j < len; j++) {
                row = sheet.createRow(j + headerRow);
                Object obj = data[i].get(j);
                if (obj == null) {
                    continue;
                }
                if (obj instanceof Map) {
                    processAsMap(columns[i], row, obj);
                }  else {
                    processAsObject(columns[i], row, obj);
                } 
            }
        }
        return wb;
    }

    @SuppressWarnings("unchecked")
    private void processAsMap(String[] columns, Row row, Object obj) {
        Cell cell;
        Map<String, Object> map = (Map<String, Object>) obj;
        if (columns.length == 0) { // show all if column not specified
            Set<String> keys = map.keySet();
            int columnIndex = 0;
            for (String key : keys) {
                cell = row.createCell(columnIndex);
                cell.setCellValue(map.get(key) + "");
                columnIndex++;
            }
        } else {
            for (int j = 0, len = columns.length; j < len; j++) {
                cell = row.createCell(j);
                cell.setCellValue(map.get(columns[j]) == null ? "" : map.get(columns[j]) + "");
            }
        }
    }

    private void processAsObject(String[] columns, Row row, Object obj) throws Exception{
        Cell cell;
        if (columns.length == 0) { // show all if column not specified
            int columnIndex = 0;
            for(Field field :obj.getClass().getDeclaredFields()){
                cell = row.createCell(columnIndex);
                cell.setCellValue(getFieldValue(field,obj)+"");
                columnIndex++;
            }
        } else {
            for (int j = 0, len = columns.length; j < len; j++) {
                Field field =obj.getClass().getDeclaredField(columns[j]);
                cell = row.createCell(j);
                cell.setCellValue(field == null ? "" : getFieldValue(field,obj) + "");
            }
        }

    }
    private String getFieldValue(Field field,Object obj) throws IllegalArgumentException, IllegalAccessException{
        field.setAccessible(true);
        Object value = field.get(obj);
        if(field.getType().getName().equals("java.util.Date"))
            value = new SimpleDateFormat(data_format).format((java.util.Date)value);
        return value+"";
    }

    public ExportExcel version(String version) {
        this.version = version;
        return this;
    }

    public ExportExcel sheetName(String sheetName) {
        this.sheetNames = new String[]{sheetName};
        return this;
    }

    public ExportExcel sheetNames(String... sheetName) {
        this.sheetNames = sheetName;
        return this;
    }

    public ExportExcel cellWidth(int cellWidth) {
        this.cellWidth = cellWidth;
        return this;
    }

    public ExportExcel headerRow(int headerRow) {
        this.headerRow = headerRow;
        return this;
    }

    public ExportExcel header(String... header) {
        this.headers = new String[][]{header};
        return this;
    }

    public ExportExcel headers(String[]... headers) {
        this.headers = headers;
        return this;
    }

    public ExportExcel column(String... column) {
        this.columns = new String[][]{column};
        return this;
    }

    public ExportExcel columns(String[]... columns) {
        this.columns = columns;
        return this;
    }

}

    private InputStream inputStream;

    public String execute()  throws IOException, Exception {
        String jsonData = request.getParameter("exportData");
        String heard = request.getParameter("heard");
        Gson gson = new Gson();
        List<Object> listOrderInfo = gson.fromJson(jsonData, new TypeToken<List<Object>>() {}.getType());
        ByteArrayOutputStream out = new ByteArrayOutputStream();
        //这里获取到我们的list数据和表头数据，直接就可以导出数据到Excel了。
        ExcelWrite.data(listOrderInfo).headers(heard.split(",")).export().write(out);
        inputStream = new ByteArrayInputStream(out.toByteArray());
        return SUCCESS;
    }

    @Override
    public void setServletRequest(HttpServletRequest request) {
        this.request = request;
    }

    public InputStream getInputStream() {
        return inputStream;
    }

    public void setInputStream(InputStream inputStream) {
        this.inputStream = inputStream;
    }

}

导入Excel

导入Excel，这里的导入Excel主要是用于解析Excel后，组装成我们的java对象。然后将Excel里面的数据插入到数据库。下面就解析Excel部分稍微贴一下代码，
这部分相对来说也还比较的简单。
有个要注意的地方，在下面代码中，在组装java对象的时候，是通过反射去获取对象属性的，具备顺序性。因此需要保持，Excel的字段和java对象的属性对应上。
当然也可以不用对应。把我们取到的参数从新set到对象的属性。也就多一个赋值的过程。没什么其他东西。

/**
 * @Title: readXlsx 
 * @Description: 解析Excel的内容
 * @param path
 * @return
 * @throws Exception 
 * @author Li Xiao Fei
 * @date 2016年8月30日
 */
private List<OutOrderVo> readContext(File file)throws Exception{
    List<OutOrderVo> listXlsx = new ArrayList<OutOrderVo>();
    InputStream is = new FileInputStream(file);
    //获取Excel工作薄
    Workbook hssfWorkbook = new XSSFWorkbook (is);
    //遍历工作薄
    for (int numSheet = 0; numSheet < hssfWorkbook.getNumberOfSheets(); numSheet++) {
        //获取工作薄
        Sheet xssfSheet = hssfWorkbook.getSheetAt(0);
        if(xssfSheet == null) continue;
        //获取表头
        Row titleRow = xssfSheet.getRow(0);
        List<String> titleList = new ArrayList<String>();
        //遍历工作薄表头
        for(int cellNum=0;cellNum<titleRow.getLastCellNum();cellNum++){
            String title = getCellValue(titleRow.getCell(cellNum));
            if(title!=null && !title.isEmpty())
                titleList.add(title);
        }
        //遍历解析工作薄订单内容
        for (int rowNum = 1; rowNum < xssfSheet.getLastRowNum()+1; rowNum++) {
            //取得解析的行
            Row contextRow = xssfSheet.getRow(rowNum);
            if(contextRow == null || contextRow.getCell(1)==null || contextRow.getCell(1).toString()=="")continue;
            OutOrderVo oov = new OutOrderVo();
            Field[] beanFiled=oov.getClass().getDeclaredFields(); 
            //利用反射机制,将Excel的内容转java对象
            for (int i = 0; i < titleList.size(); i++) {
                beanFiled[i].setAccessible(true);
                beanFiled[i].set(oov, getCellValue(contextRow.getCell(i)));
            }
            listXlsx.add(oov);
        }
    }
    return listXlsx;
}

下面这段代码，主要是用于在获取Excel表格里面的值以后，对表格的值进行格式化。

/**
 * @Title: getCellValue 
 * @Description:格式化解析的excel值
 * @param cell
 * @return 
 * @author Li Xiao Fei
 * @date 2016年8月30日
 */
private String getCellValue(Cell cell) {
    String cellValue = "";
    if (cell != null) {
        switch (cell.getCellType()) {
            case Cell.CELL_TYPE_NUMERIC:
                cellValue = String.valueOf(new DecimalFormat("#").format(cell.getNumericCellValue()));
                break;
            case Cell.CELL_TYPE_STRING:
                cellValue = cell.getStringCellValue();
                break;
            case Cell.CELL_TYPE_BOOLEAN:
                cellValue = String.valueOf(cell.getBooleanCellValue());
                break;
            case Cell.CELL_TYPE_FORMULA:
                cellValue = String.valueOf(cell.getCellFormula());
                break;
            case Cell.CELL_TYPE_BLANK:
                cellValue = "";
                break;
            case Cell.CELL_TYPE_ERROR:
                cellValue = "";
                break;
            default:
                cellValue = cell.toString().trim();
                break;
        }
    }
    return cellValue.trim();
}
